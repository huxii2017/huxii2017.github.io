import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

def savefig(fig,ax,outputfile):
    plt.savefig(outputfile,bbox_inches = 'tight')
    plt.clf()
    plt.close(fig)

# 
np.random.seed(42)

# :  (~167bp)
frag_len1 = np.random.normal(loc=167, scale=20, size=8000)

#  10bp 
x = np.linspace(100, 250, 1000)
periodicity = 1 + 0.15 * np.sin(2 * np.pi * x / 10)  # 10bp 
weights = np.interp(frag_len1, x, periodicity, left=1, right=1)
frag_len1 = frag_len1[np.random.rand(len(frag_len1)) < weights / weights.max()]

# :  (~330bp)
frag_len2 = np.random.normal(loc=330, scale=25, size=2000)

# 
noise = np.random.uniform(20, 600, size=1000)

# 
frag_lengths = np.concatenate([frag_len1, frag_len2, noise])
frag_lengths = frag_lengths[(frag_lengths > 50) & (frag_lengths < 500)]

# 
bins = np.arange(20, 601, 1)
counts, _ = np.histogram(frag_lengths, bins=bins)
percent = counts / counts.sum() * 100
midpoints = (bins[:-1] + bins[1:]) / 2

# 
plt.figure(figsize=(6,6))
plt.plot(midpoints, percent, color="tomato", linewidth=2)

plt.title("Genome-wide Fragmentation Profile of cfDNA", fontsize=16, weight='bold')
plt.xlabel("Fragment Size (bp)")
plt.ylabel("Frequency (%)")
plt.xlim(20, 600)
plt.grid(True, alpha=0.3)

# 
plt.axvline(167, color='black', linestyle="--", label="Mono-nucleosome (~167bp)")
plt.axvline(330, color='grey', linestyle="--", label="Di-nucleosome (~330bp)")
plt.legend()
plt.savefig('viz_f1',bbox_inches = 'tight')
plt.show()
